          // Make an AJAX request to the PHP script
          var xhttp = new XMLHttpRequest();
          xhttp.onreadystatechange = function() {
            if (this.readyState == 4 && this.status == 200) {
              // Parse the JSON response
              var trainees = JSON.parse(this.responseText);

              // Update the HTML content dynamically
              var table = document.getElementById("traineesTable");
              table.innerHTML = ""; // Clear existing content

              var headRow = document.createElement("tr");
              var headName = document.createElement("th");
              headName.textContent = "Name";
              var headGender = document.createElement("th");
              headGender.textContent = "Gender";
              var headGoal = document.createElement("th");
              headGoal.textContent = "Goal Type";
              headRow.appendChild(headName);
              headRow.appendChild(headGender);
              headRow.appendChild(headGoal);
              table.appendChild(headRow);

              if (trainees.length > 0) {
                for (var i = 0; i < trainees.length; i++) {
                  var row = document.createElement("tr");
                  row.className = "data_row";

                  var nameCell = document.createElement("td");
                  var nameLink = document.createElement("a");
                  nameLink.href = "remove_trainer.php?trainerId=" + <?php echo $_SESSION["user_id"]; ?> + "&traineeId=" + encodeURIComponent(trainees[i].id);
                  nameLink.textContent = trainees[i].name;
                  nameCell.appendChild(nameLink);
                  row.appendChild(nameCell);

                  var genderCell = document.createElement("td");
                  genderCell.textContent = trainees[i].gender;
                  row.appendChild(genderCell);

                  var goalCell = document.createElement("td");
                  goalCell.textContent = trainees[i].goal_type;
                  row.appendChild(goalCell);

                  table.appendChild(row);
                }
              } else {
                var emptyRow = document.createElement("tr");
                var emptyCell = document.createElement("td");
                emptyCell.colSpan = 3;
                emptyCell.className = "warn";
                emptyCell.innerHTML = "<a>You haven't added any Trainees yet</a>";
                emptyRow.appendChild(emptyCell);
                table.appendChild(emptyRow);
              }
            }
          };

          xhttp.open("GET", "fetch_trainees.php", true);
          xhttp.send();
